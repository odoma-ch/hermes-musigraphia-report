  const key = `${v.lat},${v.lng}`;
  cityCounts[key] = (cityCounts[key] || 0) + 1;
});
const cityIndex = {};

function createCircleMarker(v) {
  const cfg = typeConfig[v.type];
  const key = `${v.lat},${v.lng}`;
  cityIndex[key] = (cityIndex[key] || 0) + 1;
  const idx = cityIndex[key];
  const total = cityCounts[key];

  // Spiral offset for overlapping markers
  let latOff = 0, lngOff = 0;
  if (total > 1) {
    const angle = (2 * Math.PI * idx) / total;
    const radius = 0.04 + 0.02 * total;
    latOff = radius * Math.cos(angle) * 0.7;
    lngOff = radius * Math.sin(angle);
  }

  const lat = v.lat + latOff;
  const lng = v.lng + lngOff;

  const marker = L.circleMarker([lat, lng], {
    radius: 7, fillColor: cfg.color, fillOpacity: 0.85,
    color: "#fff", weight: 1.5, opacity: 0.6
  }).addTo(map);

  const popupContent = `
    <div class="popup-city">${cfg.icon} ${v.city}</div>
    <div class="popup-hall">${v.hall !== "—" ? v.hall : "General city venue"}</div>
    <div class="popup-coords">📍 ${v.lat.toFixed(4)}, ${v.lng.toFixed(4)}</div>
    <a class="popup-link" href="https://www.wikidata.org/wiki/${v.wiki}" target="_blank">→ ${v.wiki} on Wikidata</a>
  `;
  marker.bindPopup(popupContent);

  marker.on("mouseover", function() { this.setStyle({ radius: 10, fillOpacity: 1 }); });
  marker.on("mouseout", function() { this.setStyle({ radius: 7, fillOpacity: 0.85 }); });

  return { marker, lat, lng };
}

const markerRefs = venues.map(v => ({ ...createCircleMarker(v), venue: v }));

// Build legend
const legendEl = document.getElementById("legend");
const typeCounts = {};
venues.forEach(v => { typeCounts[v.type] = (typeCounts[v.type] || 0) + 1; });

Object.entries(typeConfig).forEach(([type, cfg]) => {
  if (!typeCounts[type]) return;
  const item = document.createElement("div");
  item.className = "legend-item";
  item.innerHTML = `
    <span class="legend-dot" style="background:${cfg.color}"></span>
    <span>${cfg.icon} ${cfg.label}</span>
    <span class="legend-count">${typeCounts[type]}</span>
  `;
  item.onclick = () => {
    const bounds = [];
    markerRefs.filter(m => m.venue.type === type).forEach(m => bounds.push([m.lat, m.lng]));
    if (bounds.length) map.fitBounds(bounds, { padding: [40, 40], maxZoom: 10 });
  };
  legendEl.appendChild(item);
});

// Build venue list
const listEl = document.getElementById("venue-list");
venues.forEach((v, i) => {
  const cfg = typeConfig[v.type];
  const entry = document.createElement("div");
  entry.className = "venue-entry";
  entry.style.borderLeftColor = cfg.color;
  entry.innerHTML = `
    <div class="city">${cfg.icon} ${v.city}</div>
    <div class="hall">${v.hall !== "—" ? v.hall : "General city venue"}</div>
    <a class="wiki-link" href="https://www.wikidata.org/wiki/${v.wiki}" target="_blank">${v.wiki}</a>
  `;
  entry.onclick = () => {
    const ref = markerRefs[i];
    map.setView([ref.lat, ref.lng], 12);
    ref.marker.openPopup();
  };
  listEl.appendChild(entry);
});
